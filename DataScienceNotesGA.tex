\newglossaryentry{task}{name=task,description=An abstract representation of a problem we want to solve regarding the domain objects of import.}
\newglossaryentry{decision rule}{name={decision rule}, description={given an observable random variable $x$ over the probability space  $(\mathcal{x},\sigma, p_\theta)$, determined by a parameter $\theta \in \theta$, and a set a of possible actions, a (deterministic) decision rule is a function $\delta : \mathcal{x} \rightarrow a$.}}
\newglossaryentry{domain objects}{name={domain objects}, description={Elements of an intuitive model prior to applying machine learning to it, \eg emails in a spam filter, movies in a rating system}}
\newglossaryentry{feature representation}{name={feature representation},description={Describe the relevant objects in our domains for use in a machine learning model.}}
\newglossaryentry{machine learning} {name={machine learning},
  description={
  \begin{enumerate}
    \item The systematic study of algorithms and systems that improve their knowledge or performance with experience.
    \item Using the right features to build the right models that achieve the right tasks
  \end{enumerate}}}
  \newglossaryentry{generalization}{name=Generalization, description={being able to use training data as experience to improve on the outcome of a given task}}
\newglossaryentry{model}{name={model},
  description={
  \begin{enumerate}
    \item A map from data points to outputs
    \item Is itself produced as the output of a machine learning algorithm applied to training data
  \end{enumerate}} }
\newglossaryentry{posterior probability}{name=posterior probability, description=probability taken after a given feature was observed}
